## Variables de entorno necesarias

```bash

SENDGRID_FROM_EMAIL=noreply@yourdomain.com
SENDGRID_FROM_NAME=ZiteBackJS Notifier
```

## Ejemplo de configuraci√≥n

```javascript
const sgMail = require('@sendgrid/mail');

// Configurar SendGrid
sgMail.setApiKey(process.env.SENDGRID_API_KEY);

// Enviar notificaci√≥n de descarga completada
async function enviarNotificacionDescarga(siteName, totalRecursos, emailDestino) {
    const msg = {
        to: emailDestino,
        from: {
            email: process.env.SENDGRID_FROM_EMAIL,
            name: process.env.SENDGRID_FROM_NAME
        },
        subject: `‚úÖ ZiteBackJS - Descarga completada: ${siteName}`,
        html: `
            <h2>üéâ Descarga completada exitosamente</h2>
            <p><strong>Sitio:</strong> ${siteName}</p>
            <p><strong>Recursos descargados:</strong> ${totalRecursos}</p>
            <p><strong>Hora de finalizaci√≥n:</strong> ${new Date().toLocaleString()}</p>
            <hr>
            <p><em>ZiteBackJS v3.7.8 - Sistema Revolucionario de URLs sin Protocolo</em></p>
        `
    };

    try {
        await sgMail.send(msg);
        console.log('üìß Notificaci√≥n enviada correctamente');
    } catch (error) {
        console.error('‚ùå Error enviando notificaci√≥n:', error);
    }
}

module.exports = { enviarNotificacionDescarga };
```
